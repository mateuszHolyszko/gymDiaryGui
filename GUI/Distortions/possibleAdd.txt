// === Rolling noisy horizontal band with falloff, only bright spots drawn ===
    float band_speed = 0.5;    // speed of the roll
    float band_height = 0.05;  // total vertical size of the band
    float band_pos = fract(time * band_speed); // center position in [0,1]

    // distance from pixel to band center
    float dist_to_band = abs(uv.y - band_pos);

    // smooth vertical falloff
    float edge_fade = 1.0 - smoothstep(band_height * 0.5, band_height * 0.5 + 0.02, dist_to_band);

    if (edge_fade > 0.0) {
        // Per-pixel random noise (changes over time)
        float pixelNoise = rand(vec2(floor(uv.x * 320.0), floor(uv.y * 480.0) + floor(time * 60.0)));
        
        // Random gaps per row (makes spotty lines)
        float linePattern = step(0.7, rand(vec2(0.0, floor(uv.y * 480.0) + time * 15.0)));

        // Combine to get noisy brightness
        float bandValue = pixelNoise * linePattern;

        // Threshold â€” only draw if bright enough
        if (bandValue > 0.2) {
            vec3 bandColor = vec3(1.0, 0.98, 0.95) * bandValue;
            color = mix(color, bandColor, edge_fade); // override only bright spots
        }
    }
